                                                                                            LPS JAVA Solution using HASHING

                                                                                            static int lps(String S)
                                                                                            {
                                                                                                long $prefix=0L, $suffix=0L;
                                                                                                char[] C = S.toCharArray();
                                                                                                long INF = 1000000007L;
                                                                                                int N = S.length()-1;
                                                                                                long ML = 1L;
                                                                                                int LPS = 0;
                                                                                                    
                                                                                                for (int i=0, j=N; i < N; i++, j--)
                                                                                                {
                                                                                                    char pX = C[i], sX = C[j];
                                                                                                    $prefix *= 26;	$prefix += pX;	$prefix %= INF;
                                                                                                    $suffix += (ML * sX);	$suffix %= INF;
                                                                                                    ML *= 26;	ML %= INF;
                                                                                                    if ($prefix == $suffix)
                                                                                                        LPS = i+1;
                                                                                                }
                                                                                                return LPS;
                                                                                            }
